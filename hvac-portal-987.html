function calculateProfit() { // 1. INPUT CAPTURE & HARD CONSTRAINT FALLBACKS (PRD Rule: Handle non-numeric input as 0) const jobPrice = Math.max(0, parseFloat(document.getElementById('job-price').value) || 0); const commissionType = document.getElementById('commission-type').value; // "%" or "Flat" const commissionValue = Math.max(0, parseFloat(document.getElementById('commission-value').value) || 0); const costEquipment = Math.max(0, parseFloat(document.getElementById('cost-equipment').value) || 0); const costMaterials = Math.max(0, parseFloat(document.getElementById('cost-materials').value) || 0); const costSubcontractor = Math.max(0, parseFloat(document.getElementById('cost-subcontractor').value) || 0); const techHours = Math.max(0, parseFloat(document.getElementById('tech-hours').value) || 0); const hourlyRate = Math.max(0, parseFloat(document.getElementById('hourly-rate').value) || 0); const driveTime = Math.max(0, parseFloat(document.getElementById('drive-time').value) || 0); const fleetCost = Math.max(0, parseFloat(document.getElementById('fleet-cost').value) || 0); const overheadType = document.getElementById('overhead-type').value; // "%" or "Flat" const overheadValue = Math.max(0, parseFloat(document.getElementById('overhead-value').value) || 0); // 2. CORE PRD ENGINE MATH (Source of Truth calculations) // Commission Paid let commissionPaid = 0; if (commissionType === "%") { commissionPaid = jobPrice * (commissionValue / 100); } else { commissionPaid = commissionValue; } // Direct Costs const directCosts = costEquipment + costMaterials + costSubcontractor + (techHours * hourlyRate) + commissionPaid; // Overhead Allocation Cost let overheadAllocation = 0; if (overheadType === "%") { overheadAllocation = jobPrice * (overheadValue / 100); } else { overheadAllocation = overheadValue; } const overheadCosts = (driveTime * fleetCost) + overheadAllocation; // Net Profit Output let netProfit = jobPrice - directCosts - overheadCosts; // Gross Margin % let grossMarginPercent = 0; if (jobPrice > 0) { grossMarginPercent = ((jobPrice - directCosts) / jobPrice) * 100; } // Net Margin % let netMarginPercent = 0; if (jobPrice > 0) { netMarginPercent = (netProfit / jobPrice) * 100; } // PRD Anti-Crash Rule: If calculations result in NaN or Infinity, reset to 0 if (isNaN(netProfit) || !isFinite(netProfit)) netProfit = 0; if (isNaN(grossMarginPercent) || !isFinite(grossMarginPercent)) grossMarginPercent = 0; if (isNaN(netMarginPercent) || !isFinite(netMarginPercent)) netMarginPercent = 0; // 3. PERSISTENCE LAYER (PRD Rule: Use browser-local storage to cache input values instantly) const inputCache = { jobPrice, commissionType, commissionValue, costEquipment, costMaterials, costSubcontractor, techHours, hourlyRate, driveTime, fleetCost, overheadType, overheadValue }; localStorage.setItem('hvac_calculator_cache', JSON.stringify(inputCache)); // 4. DISPLAY LAYER RENDERING (Precision: Rounding to 2 decimal places done strictly here) document.getElementById('lbl-net-profit').textContent = (netProfit >= 0 ? '$' : '-$') + Math.abs(netProfit).toLocaleString('en-US', { minimumFractionDigits: 2, maximumFractionDigits: 2 }); document.getElementById('lbl-gross-margin').textContent = grossMarginPercent.toFixed(2) + '%'; document.getElementById('lbl-net-margin').textContent = netMarginPercent.toFixed(2) + '%'; // 5. PRD UI SUMMARY CARD HIGHLIGHTS ENGINE const summaryCard = document.getElementById('summary-card'); // Target container const metricText = document.querySelectorAll('.summary-metric-value'); // Target text elements for clear visibility if (netMarginPercent < 10) { // RED ALERT PROFILE: High risk/unprofitable threshold summaryCard.style.backgroundColor = 'rgba(239, 68, 68, 0.15)'; // Muted Dark Red border/bg tint summaryCard.style.borderColor = '#ef4444'; metricText.forEach(el => el.style.color = '#ef4444'); } else if (netMarginPercent < 20) { // AMBER ALERT PROFILE: Cautionary target threshold summaryCard.style.backgroundColor = 'rgba(245, 158, 11, 0.15)'; // Muted Dark Amber border/bg tint summaryCard.style.borderColor = '#f59e0b'; metricText.forEach(el => el.style.color = '#f59e0b'); } else { // GREEN TARGET PROFILE: Healthy operational safety metrics summaryCard.style.backgroundColor = 'rgba(34, 197, 94, 0.15)'; // Muted Dark Green border/bg tint summaryCard.style.borderColor = '#22c55e'; metricText.forEach(el => el.style.color = '#22c55e'); } } // 6. CACHE RETRIEVAL INITIALIZER (Restores inputs instantly when page refreshes) window.addEventListener('DOMContentLoaded', () => { const savedCache = localStorage.getItem('hvac_calculator_cache'); if (savedCache) { const cache = JSON.parse(savedCache); if (document.getElementById('job-price')) document.getElementById('job-price').value = cache.jobPrice; if (document.getElementById('commission-type')) document.getElementById('commission-type').value = cache.commissionType; if (document.getElementById('commission-value')) document.getElementById('commission-value').value = cache.commissionValue; if (document.getElementById('cost-equipment')) document.getElementById('cost-equipment').value = cache.costEquipment; if (document.getElementById('cost-materials')) document.getElementById('cost-materials').value = cache.costMaterials; if (document.getElementById('cost-subcontractor')) document.getElementById('cost-subcontractor').value = cache.costSubcontractor; if (document.getElementById('tech-hours')) document.getElementById('tech-hours').value = cache.techHours; if (document.getElementById('hourly-rate')) document.getElementById('hourly-rate').value = cache.hourlyRate; if (document.getElementById('drive-time')) document.getElementById('drive-time').value = cache.driveTime; if (document.getElementById('fleet-cost')) document.getElementById('fleet-cost').value = cache.fleetCost; if (document.getElementById('overhead-type')) document.getElementById('overhead-type').value = cache.overheadType; if (document.getElementById('overhead-value')) document.getElementById('overhead-value').value = cache.overheadValue; } // Perform initial layout calculations on load calculateProfit(); });
